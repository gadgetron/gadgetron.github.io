108  template<class T> void
109  permute( hoNDArray<T> *in, hoNDArray<T> *out, std::vector<size_t> *dim_order, int shift_mode = 0)
110  {
111  if( in == 0x0 || out == 0x0 || dim_order == 0x0 ) {
112  throw std::runtime_error("permute(): invalid pointer provided");;
113  }
114 
115  if( in == out ){
116  throw std::runtime_error("permute(): in-place permutation not supported");;
117  }
118 
119  // Check ordering array
120  if (dim_order->size() > in->get_number_of_dimensions()) {
121  throw std::runtime_error("hoNDArray::permute - Invalid length of dimension ordering array");;
122  }
123 
124  std::vector<size_t> dim_count(in->get_number_of_dimensions(),0);
125  for (size_t i = 0; i < dim_order->size(); i++) {
126  if ((*dim_order)[i] >= in->get_number_of_dimensions()) {
127  throw std::runtime_error("hoNDArray::permute - Invalid dimension order array");;
128  }
129  dim_count[(*dim_order)[i]]++;
130  }
131 
132  // Create an internal array to store the dimensions
133  std::vector<size_t> dim_order_int;
134 
135  // Check that there are no duplicate dimensions
136  for (size_t i = 0; i < dim_order->size(); i++) {
137  if (dim_count[(*dim_order)[i]] != 1) {
138  throw std::runtime_error("hoNDArray::permute - Invalid dimension order array (duplicates)");;
139 
140  }
141  dim_order_int.push_back((*dim_order)[i]);
142  }
143 
144  for (size_t i = 0; i < dim_order_int.size(); i++) {
145  if ((*in->get_dimensions())[dim_order_int[i]] != out->get_size(i)) {
146  throw std::runtime_error("permute(): dimensions of output array do not match the input array");;
147  }
148  }
149 
150  // Pad dimension order array with dimension not mentioned in order array
151  if (dim_order_int.size() < in->get_number_of_dimensions()) {
152  for (size_t i = 0; i < dim_count.size(); i++) {
153  if (dim_count[i] == 0) {
154  dim_order_int.push_back(i);
155  }
156  }
157  }
158 
159  T* o = out->get_data_ptr();
160 
161  // if memcpy can be used during permute
162  size_t stride = 1;
163  size_t num_dim_memcpy = 0;
164  for (size_t i = 0; i < dim_order_int.size(); i++) {
165  if (dim_order_int[i]==i){
166  stride *= in->get_size(i);
167  num_dim_memcpy = i;
168  }
169  else{
170  break;
171  }
172  }
173 
174  if (stride == 1) {
175  // point by point assignment is needed
176  ArrayIterator it(in->get_dimensions().get(), &dim_order_int);
177  for (size_t i = 0; i < in->get_number_of_elements(); i++) {
178  o[i] = in->get_data_ptr()[it.get_current_idx()];
179  it.advance();
180  }
181  }
182  else {
183  // memcpy can be used
184 
185  size_t nDim = in->get_number_of_dimensions();
186  size_t num_memcpy = in->get_number_of_elements() / stride;
187 
188  if (num_dim_memcpy == nDim - 1){
189  memcpy(out->begin(), in->begin(), in->get_number_of_bytes());
190  return;
191  }
192 
193  // for the array index calculation
194  std::vector<size_t> dim_permute(nDim-num_dim_memcpy-1);
195  for (size_t i = num_dim_memcpy+1; i < dim_order_int.size(); i++) {
196  dim_permute[i - num_dim_memcpy - 1] = in->get_size(i);
197  }
198 
199  size_t n;
200 
201  hoNDArray<T> permuteArray(dim_permute, in->begin(), false);
202 
203  // starting index for in and out array for every permute memcpy operation
204  std::vector<size_t> ind_permute_in(dim_permute.size(), 0), ind_in(nDim, 0), ind_out(nDim, 0);
205 
206  for (n = 0; n < num_memcpy; n++) {
207  permuteArray.calculate_index(n, ind_permute_in);
208  memcpy(&ind_in[0] + num_dim_memcpy + 1, &ind_permute_in[0], sizeof(size_t)*ind_permute_in.size());
209 
210  // permute the indexes
211  for (size_t i = 0; i < nDim; i++) {
212  ind_out[i] = ind_in[dim_order_int[i]];
213  }
214 
215  size_t offset_in = in->calculate_offset(ind_in);
216  size_t offset_out = out->calculate_offset(ind_out);
217 
218  memcpy(o + offset_out, in->begin() + offset_in, sizeof(T)*stride);
219  }
220  }
